n = int(input("Enter a number: "))

    print(f"{n} is not a prime number")
else:
    prime = True

## 17. Calculadora simple

```py
print("github.com/rossxrio\nSimple calc.")
def sum(x, y):
    return x + y
def sub(x, y):
    return x - y
def mult(x, y):
    return x * y
def div(x, y):
    return max(x, y) / min(x, y)

x = input('Insert the first number')
y = input('Insert the second number')
operation = input("What operation you wish to perform")

match operation:
    case '+':
        print(sum(int(x), int(y)))
    case '-':
        print(sub(int(x), int(y)))
    case '*':
        print(mult(int(x), int(y)))
    case '/':
        print(div(int(x), int(y)))
```

## 18. Conversión de temperaturas

```py
print("github.com/rossxrio\nTemperature Converter")

value = float(input("Enter the temperature value: "))
unitFrom = input("Enter the original unit: ").upper()
unitTo = input("Enter the unit to convert to: ").upper()

# Convert input to Celsius first
if unitFrom == "C":
    celsius = value
elif unitFrom == "F":
    celsius = (value - 32) * 5/9
elif unitFrom == "K":
    celsius = value - 273.15
else:
    print("Invalid original unit.")
    exit()

if unitTo == "C":
    result = celsius
elif unitTo == "F":
    result = celsius * 9/5 + 32
elif unitTo == "K":
    result = celsius + 273.15
else:
    print("Invalid target unit.")
    exit()

print(f"{value}°{unitFrom} is {result:.2f}°{unitTo}")
```
