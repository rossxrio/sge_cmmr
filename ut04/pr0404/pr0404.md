## 4.- Diccionario de listas

```py
print("github.com/rossxrio\n")
asignaturas = {...}

def listar(asignatura):
    return asignaturas.get(asignatura)

def dar_baja(asignatura, estudiante):
    return asignaturas.get(asignatura).remove(estudiante)

def matricular(asignatura, estudiante):
    return asignaturas.get(asignatura).append(estudiante)

while True:
    print("(1)Listar, (2)Dar de baja, (3)Matricular, (4)Salir")
    opcion = input()
    match int(opcion):
        case 1: print(listar(input("asignatura: ")))
        case 2: print(dar_baja(input("asignatura: "), input("estudiante: ")))
        case 3: print(matricular(input("asignatura: "), input("estudiante: ")))
        case 4: break

```

## 5.- Diccionario invertido

```py
print("github.com/rossxrio\nyranoitciD")
dict = {
    "Toyota" : "Marca",
    "Blanco" : "Color"
}

def reverse_dict(d):
    return {v : k for k, v in d.items()}

print(reverse_dict(dict))

```

## 8.- Anidación de diccionarios

```py
print("github.com/rossxrio\nNested dicitionaries")

departamentos = {...}

while True:
    print("(1) List employees, (2) Add employee, (3) Delete employee, (4) Exit")
    option = int(input("option: "))
    match option:
        case 1:
            print(departamentos.get(input("insert the departament name: ")))
        case 2:
            departamentos.get(input("insert the departament name: ")).update({input("Employee name: ") : input("Role: ")})
        case 3:
            departamentos.get(input("insert the departament name: ")).pop(input("Employee name: "))
        case 4:
            break
```

## 9.- Transformación de datos

```py
from collections import defaultdict
print("github.com/rossxrio\nNested dicitionaries")

estudiantes = {...}
estudiantes_v = list(estudiantes.values())
students_grades = {k : sum(list(estudiantes.get(k).values())) / 3 for k in estudiantes.keys()}


sub = defaultdict(float)
for i in range(len(estudiantes_v)):
    k = ""
    for k in estudiantes_v[i].keys():
        sub[k]+= estudiantes_v[i].get(k)

sub.update({k : sub.get(k) / (6) for k in sub})

print(f"Grades per student {students_grades}")
print(f"Grades per subject {sub}")
```
